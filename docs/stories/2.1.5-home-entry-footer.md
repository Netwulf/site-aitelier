## Updated Navigation

**Minimal navigation for new structure:**

```tsx
// src/components/NavigationV2.tsx

const navItems = [
  { label: 'O Campo', href: '/campo' },
  { label: 'Estudos', href: '/estudos' },
  { label: 'Studio', href: '/studio' },
  { label: 'Archive', href: '/archive' },
];

export const NavigationV2 = () => {
  const [scrolled, setScrolled] = useState(false);

  useEffect(() => {
    const handleScroll = () => setScrolled(window.scrollY > 100);
    window.addEventListener('scroll', handleScroll);
    return () => window.removeEventListener('scroll', handleScroll);
  }, []);

  return (
    <header className={cn(
      "fixed top-0 left-0 right-0 z-50 transition-all duration-300",
      scrolled
        ? "bg-brutal-black/90 backdrop-blur-sm border-b border-text-muted/10"
        : "bg-transparent"
    )}>
      <nav className="max-w-container mx-auto px-6 py-4
                      flex justify-between items-center">
        <a href="/" className="font-display text-brutal-white text-lg">
          AI.TELIER
        </a>

        <div className="hidden md:flex items-center gap-8">
          {navItems.map((item) => (
            <a
              key={item.href}
              href={item.href}
              className="text-sm text-text-secondary hover:text-brutal-white
                        transition-colors font-mono"
            >
              {item.label}
            </a>
          ))}
        </div>

        {/* Mobile menu button */}
        <button className="md:hidden text-brutal-white">
          <span className="font-mono text-sm">[Menu]</span>
        </button>
      </nav>
    </header>
  );
};
```

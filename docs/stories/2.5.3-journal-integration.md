## Implementation

```tsx
// src/components/JournalList.tsx

interface JournalEntry {
  slug: string;
  title: string;
  date: string;
  readingTime: string;
  excerpt?: string;
}

// This could come from a CMS or markdown files
const journalEntries: JournalEntry[] = [
  {
    slug: 'marca-nao-precisa-conteudo',
    title: 'Por que sua marca não precisa de mais conteúdo',
    date: '2025-01-15',
    readingTime: '5 min',
  },
  {
    slug: 'problema-personal-branding',
    title: 'O problema com personal branding',
    date: '2025-01-10',
    readingTime: '7 min',
  },
  {
    slug: 'ia-como-camera',
    title: 'IA como câmera: um manifesto',
    date: '2025-01-05',
    readingTime: '10 min',
  },
  {
    slug: 'futuro-ancestral-significado',
    title: 'Futuro Ancestral: o que isso significa',
    date: '2024-12-28',
    readingTime: '8 min',
  },
  {
    slug: 'morte-founder-invisivel',
    title: 'A morte do founder invisível',
    date: '2024-12-20',
    readingTime: '6 min',
  },
];

export const JournalList = () => {
  return (
    <div>
      <div className="mb-12">
        <p className="text-text-secondary text-lg">
          Pensamento em processo.
        </p>
      </div>

      <div className="space-y-0 border-t border-text-muted/20">
        {journalEntries.map((entry, index) => (
          <JournalEntryRow key={entry.slug} entry={entry} index={index} />
        ))}
      </div>
    </div>
  );
};

const JournalEntryRow = ({
  entry,
  index,
}: {
  entry: JournalEntry;
  index: number;
}) => {
  const formattedDate = new Date(entry.date).toLocaleDateString('pt-BR', {
    day: '2-digit',
    month: 'short',
    year: 'numeric',
  });

  return (
    <motion.a
      href={`/journal/${entry.slug}`}
      initial={{ opacity: 0, y: 10 }}
      animate={{ opacity: 1, y: 0 }}
      transition={{ delay: index * 0.05 }}
      className="block border-b border-text-muted/20 py-6
                 hover:bg-stone-dark/30 transition-colors
                 group"
    >
      <div className="flex flex-col md:flex-row md:items-center
                      md:justify-between gap-2">
        {/* Title */}
        <h3 className="font-display text-lg text-brutal-white
                       group-hover:text-ancestral-amber transition-colors">
          {entry.title}
        </h3>

        {/* Meta */}
        <div className="flex items-center gap-4 text-sm text-text-muted
                        font-mono">
          <span>{formattedDate}</span>
          <span>·</span>
          <span>{entry.readingTime}</span>
        </div>
      </div>
    </motion.a>
  );
};
```

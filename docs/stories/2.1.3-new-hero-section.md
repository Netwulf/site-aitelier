## Component Implementation

```tsx
// src/components/HeroV2.tsx

import { motion } from 'framer-motion';

interface HeroV2Props {
  variant?: 'A' | 'B' | 'C';
}

const heroContent = {
  A: {
    lines: ['O futuro é ancestral.'],
    className: 'text-hero'
  },
  B: {
    lines: ['A técnica virou commodity.', 'O invisível virou o centro.'],
    className: 'text-display-1'
  },
  C: {
    lines: ['Um campo para quem cria', 'na era da inteligência artificial.'],
    className: 'text-display-1'
  }
};

export const HeroV2 = ({ variant = 'A' }: HeroV2Props) => {
  const content = heroContent[variant];

  return (
    <section className="relative min-h-screen flex items-center justify-center
                        bg-brutal-black overflow-hidden">
      {/* Grain overlay */}
      <div className="absolute inset-0 opacity-[0.03] pointer-events-none"
           style={{
             backgroundImage: `url("data:image/svg+xml,%3Csvg viewBox='0 0 256 256' xmlns='http://www.w3.org/2000/svg'%3E%3Cfilter id='noise'%3E%3CfeTurbulence type='fractalNoise' baseFrequency='0.9' numOctaves='4' stitchTiles='stitch'/%3E%3C/filter%3E%3Crect width='100%25' height='100%25' filter='url(%23noise)'/%3E%3C/svg%3E")`
           }}
      />

      {/* Content */}
      <div className="relative z-10 px-6 text-center max-w-4xl mx-auto">
        <motion.div
          initial={{ opacity: 0, y: 20 }}
          animate={{ opacity: 1, y: 0 }}
          transition={{ duration: 0.8, ease: [0.25, 0.1, 0.25, 1] }}
        >
          {content.lines.map((line, index) => (
            <motion.h1
              key={index}
              className={cn(
                "font-display font-medium text-brutal-white leading-tight",
                content.className,
                index > 0 && "mt-2"
              )}
              initial={{ opacity: 0, y: 20 }}
              animate={{ opacity: 1, y: 0 }}
              transition={{
                duration: 0.8,
                delay: index * 0.15,
                ease: [0.25, 0.1, 0.25, 1]
              }}
            >
              {line}
            </motion.h1>
          ))}
        </motion.div>
      </div>

      {/* Scroll indicator */}
      <motion.div
        className="absolute bottom-8 left-1/2 -translate-x-1/2"
        initial={{ opacity: 0 }}
        animate={{ opacity: 1 }}
        transition={{ delay: 1.5, duration: 0.6 }}
      >
        <motion.div
          animate={{ y: [0, 8, 0] }}
          transition={{ repeat: Infinity, duration: 2, ease: "easeInOut" }}
          className="text-text-muted text-sm font-mono"
        >
          ↓
        </motion.div>
      </motion.div>
    </section>
  );
};
```

## Animation Specs

| Element | Animation | Duration | Delay | Easing |
|---------|-----------|----------|-------|--------|
| Hero text | Fade + translateY | 0.8s | 0s | cubic-bezier(0.25, 0.1, 0.25, 1) |
| Second line (if B/C) | Fade + translateY | 0.8s | 0.15s | same |
| Scroll indicator | Fade in | 0.6s | 1.5s | ease |
| Scroll arrow | Bounce | 2s loop | - | easeInOut |

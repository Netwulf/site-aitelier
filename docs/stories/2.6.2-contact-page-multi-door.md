## Implementation

```tsx
// src/pages/Contact.tsx (rewrite)

import { motion } from 'framer-motion';
import { NavigationV2 } from '@/components/NavigationV2';
import { FooterV2 } from '@/components/FooterV2';
import { SectionContainer } from '@/components/SectionContainer';
import { staggerContainer, staggerItem } from '@/utils/motionVariants';

interface EntryDoor {
  id: string;
  title: string;
  description: string;
  cta: string;
  href: string;
  type: 'link' | 'form';
}

const entryDoors: EntryDoor[] = [
  {
    id: 'estudos',
    title: 'ESTUDOS',
    description: 'Quero aprender a linguagem',
    cta: 'Entrar na lista de espera',
    href: '#estudos-form',
    type: 'form',
  },
  {
    id: 'studio',
    title: 'STUDIO',
    description: 'Quero trabalhar com vocês',
    cta: 'Iniciar conversa',
    href: '#studio-form',
    type: 'form',
  },
  {
    id: 'conselheiros',
    title: 'CONSELHEIROS',
    description: 'Quero acessar as mentes',
    cta: 'Conhecer os conselheiros',
    href: '/campo#conselheiros',
    type: 'link',
  },
  {
    id: 'newsletter',
    title: 'APENAS ACOMPANHAR',
    description: 'Quero receber os ensaios',
    cta: 'Entrar para a newsletter',
    href: '#newsletter-form',
    type: 'form',
  },
];

const Contact = () => {
  return (
    <div className="min-h-screen bg-brutal-black">
      <NavigationV2 />

      <main>
        {/* Hero */}
        <section className="pt-32 pb-8 px-6">
          <div className="max-w-container mx-auto">
            <motion.h1
              initial={{ opacity: 0, y: 20 }}
              animate={{ opacity: 1, y: 0 }}
              className="font-display text-display-1 text-brutal-white mb-8"
            >
              Entrar no Campo
            </motion.h1>

            <motion.div
              initial={{ opacity: 0 }}
              animate={{ opacity: 1 }}
              transition={{ delay: 0.2 }}
              className="max-w-xl space-y-4 text-lg"
            >
              <p className="text-text-secondary">
                AI.TELIER não é para todo mundo.
                <br />
                E não deveria ser.
              </p>
              <p className="text-brutal-white">
                Se você chegou até aqui,
                <br />
                uma dessas portas pode fazer sentido:
              </p>
            </motion.div>
          </div>
        </section>

        {/* Entry Doors */}
        <SectionContainer>
          <motion.div
            variants={staggerContainer}
            initial="hidden"
            animate="visible"
            className="grid grid-cols-1 md:grid-cols-2 gap-4 max-w-4xl"
          >
            {entryDoors.map((door) => (
              <EntryDoorCard key={door.id} door={door} />
            ))}
          </motion.div>
        </SectionContainer>

        {/* Contact Info */}
        <SectionContainer className="py-16">
          <div className="flex flex-col md:flex-row gap-8 text-text-muted
                          font-mono text-sm">
            <a
              href="mailto:contato@ai.telier"
              className="hover:text-brutal-white transition-colors"
            >
              contato@ai.telier
            </a>
            <a
              href="https://instagram.com/ai.telier"
              target="_blank"
              rel="noopener noreferrer"
              className="hover:text-brutal-white transition-colors"
            >
              @ai.telier
            </a>
          </div>
        </SectionContainer>
      </main>

      <FooterV2 />
    </div>
  );
};

const EntryDoorCard = ({ door }: { door: EntryDoor }) => {
  return (
    <motion.div
      variants={staggerItem}
      className="p-8 border border-text-muted/20
                 hover:border-text-muted/40 transition-colors
                 bg-stone-dark/30"
    >
      <h3 className="font-mono text-sm tracking-widest text-text-muted mb-4">
        {door.title}
      </h3>

      <p className="text-lg text-brutal-white mb-6">
        {door.description}
      </p>

      {door.type === 'link' ? (
        <a
          href={door.href}
          className="inline-block text-sm font-mono text-ancestral-amber
                     hover:text-brutal-white transition-colors"
        >
          [{door.cta}] →
        </a>
      ) : (
        <button
          onClick={() => {
            // Scroll to form or open modal
            document.querySelector(door.href)?.scrollIntoView({
              behavior: 'smooth'
            });
          }}
          className="text-sm font-mono text-ancestral-amber
                     hover:text-brutal-white transition-colors"
        >
          [{door.cta}]
        </button>
      )}
    </motion.div>
  );
};

export default Contact;
```

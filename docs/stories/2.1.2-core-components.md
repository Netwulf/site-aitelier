## Components to Create

### 1. SectionNumber

Displays section number in margin (01, 02, 03...)

```tsx
// src/components/SectionNumber.tsx

interface SectionNumberProps {
  number: number;
  className?: string;
}

export const SectionNumber = ({ number, className }: SectionNumberProps) => {
  return (
    <span className={cn(
      "font-mono text-sm text-text-muted tracking-widest",
      className
    )}>
      {number.toString().padStart(2, '0')}
    </span>
  );
};
```

**Usage:**
```tsx
<SectionNumber number={1} />
// Renders: 01
```

### 2. SectionContainer

Wrapper for page sections with consistent spacing and optional number.

```tsx
// src/components/SectionContainer.tsx

interface SectionContainerProps {
  children: React.ReactNode;
  number?: number;
  className?: string;
  id?: string;
}

export const SectionContainer = ({
  children,
  number,
  className,
  id
}: SectionContainerProps) => {
  return (
    <section
      id={id}
      className={cn(
        "relative py-section px-gutter max-w-container mx-auto",
        className
      )}
    >
      {number && (
        <div className="absolute left-4 top-8 hidden lg:block">
          <SectionNumber number={number} />
        </div>
      )}
      <div className="lg:pl-16">
        {children}
      </div>
    </section>
  );
};
```

### 3. Triptych

Three-column card layout for pillars (Estudos, Studio, Archive).

```tsx
// src/components/Triptych.tsx

interface TriptychItem {
  title: string;
  subtitle: string;
  href: string;
}

interface TriptychProps {
  items: TriptychItem[];
}

export const Triptych = ({ items }: TriptychProps) => {
  return (
    <div className="grid grid-cols-1 md:grid-cols-3 gap-px bg-text-muted/20">
      {items.map((item, index) => (
        <a
          key={item.title}
          href={item.href}
          className="group bg-brutal-black p-8 md:p-12
                     border-brutal hover:border-brutal-strong
                     transition-all duration-300"
        >
          <SectionNumber number={index + 1} className="mb-4 block" />
          <h3 className="font-display text-display-2 text-brutal-white mb-4
                         group-hover:text-ancestral-amber transition-colors">
            {item.title}
          </h3>
          <p className="text-text-secondary text-sm leading-relaxed">
            {item.subtitle}
          </p>
          <div className="mt-6 text-text-muted group-hover:text-brutal-white
                          transition-colors text-sm">
            [Entrar] →
          </div>
        </a>
      ))}
    </div>
  );
};
```

**Usage:**
```tsx
<Triptych items={[
  { title: "ESTUDOS", subtitle: "Onde se aprende a linguagem", href: "/estudos" },
  { title: "STUDIO", subtitle: "Onde o campo encontra o mundo", href: "/studio" },
  { title: "ARCHIVE", subtitle: "Onde o trabalho permanece", href: "/archive" },
]} />
```

### 5. TextBlock

Styled text container for body copy.

```tsx
// src/components/TextBlock.tsx

interface TextBlockProps {
  children: React.ReactNode;
  size?: 'sm' | 'md' | 'lg';
  className?: string;
}

export const TextBlock = ({
  children,
  size = 'md',
  className
}: TextBlockProps) => {
  const sizeClasses = {
    sm: 'text-sm leading-relaxed',
    md: 'text-base leading-relaxed',
    lg: 'text-lg md:text-xl leading-relaxed',
  };

  return (
    <div className={cn(
      "text-text-secondary space-y-4",
      sizeClasses[size],
      className
    )}>
      {children}
    </div>
  );
};
```

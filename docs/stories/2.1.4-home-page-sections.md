### Section 4: Visual Archive Preview

**Specs:**
- Grid of 6-9 images
- Cinematic treatment (grain, contrast)
- Links to /archive
- No captions

**Implementation:**

```tsx
// src/components/ArchivePreview.tsx

const archiveImages = [
  '/assets/archive/img-1.jpg',
  '/assets/archive/img-2.jpg',
  '/assets/archive/img-3.jpg',
  '/assets/archive/img-4.jpg',
  '/assets/archive/img-5.jpg',
  '/assets/archive/img-6.jpg',
];

export const ArchivePreview = () => {
  return (
    <SectionContainer number={4}>
      <div className="flex justify-between items-end mb-8">
        <div>
          <h2 className="font-mono text-sm tracking-widest text-text-muted mb-2">
            VISUAL ARCHIVE
          </h2>
          <p className="text-lg text-text-secondary">
            Imagem, Cinema e Obra
          </p>
        </div>
        <a href="/archive"
           className="text-sm text-text-muted hover:text-brutal-white
                      transition-colors font-mono">
          [Ver tudo] →
        </a>
      </div>

      <div className="grid grid-cols-2 md:grid-cols-3 gap-1">
        {archiveImages.map((src, index) => (
          <motion.div
            key={src}
            initial={{ opacity: 0 }}
            whileInView={{ opacity: 1 }}
            viewport={{ once: true }}
            transition={{ delay: index * 0.1 }}
            className="aspect-[16/9] relative overflow-hidden group"
          >
            <img
              src={src}
              alt=""
              className="w-full h-full object-cover
                        filter contrast-110 saturate-90
                        group-hover:scale-105 transition-transform duration-500"
            />
            <div className="absolute inset-0 bg-brutal-black/20
                           group-hover:bg-transparent transition-colors" />
          </motion.div>
        ))}
      </div>
    </SectionContainer>
  );
};
```

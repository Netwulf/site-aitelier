### Portfolio Data Hook

```typescript
// src/hooks/usePortfolioData.ts
import { useTranslation } from 'react-i18next';
import { portfolioConfig } from '@/data/portfolioConfig';

export interface PortfolioProject {
  id: string;
  title: string;
  description: string;
  type: { short: string; long: string };
  date: string;
  location: string;
  images: Array<{
    src: string;
    alt: string;
    aspectRatio: string;
  }>;
}

export function usePortfolioData(): PortfolioProject[] {
  const { t } = useTranslation('portfolio');

  return portfolioConfig.map((project) => ({
    ...project,
    title: t(`${project.id}.title`),
    description: t(`${project.id}.description`),
    type: {
      short: t(`${project.id}.type.short`),
      long: t(`${project.id}.type.long`),
    },
    location: t(`${project.id}.location`),
    images: project.images.map((img, idx) => ({
      ...img,
      alt: t(`${project.id}.images.${idx + 1}`, { defaultValue: '' }),
    })),
  }));
}
```
